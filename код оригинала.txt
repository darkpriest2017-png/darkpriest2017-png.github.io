def strip_gigachat_disclaimer(text):
    if not text:
        return text
    lower = text.lower()
    earliest_idx = len(text)
    found = False
    for marker in GIGACHAT_DISCLAIMERS:
        idx = lower.find(marker)
        if idx != -1 and idx < earliest_idx:
            earliest_idx = idx
            found = True
    if found:
        clean = text[:earliest_idx].strip()
        clean = clean.rstrip(".,!?;: \n")
        log(
            "VALID",
            f"Disclaimer stripped at "
            f"pos {earliest_idx}"
        )
        if clean:
            return clean
        else:
            return ""
    return text

def has_disclaimer(text):
    if not text:
        return False
    lower = text.lower()
    return any(
        m in lower
        for m in GIGACHAT_DISCLAIMERS
    )

def log(level, message):
    try:
        timestamp = datetime.now().strftime(
            "%Y-%m-%d %H:%M:%S"
        )
        line = (
            f"[{timestamp}] [{level}] "
            f"{message}\n"
        )
        with open(
            LOG_FILE, "a",
            encoding="utf-8"
        ) as f:
            f.write(line)
        if os.path.exists(LOG_FILE):
            size = os.path.getsize(LOG_FILE)
            if size > 5 * 1024 * 1024:
                with open(
                    LOG_FILE, "r",
                    encoding="utf-8"
                ) as f:
                    lines = f.readlines()
                with open(
                    LOG_FILE, "w",
                    encoding="utf-8"
                ) as f:
                    f.writelines(
                        lines[-1000:]
                    )
    except Exception:
        pass

def estimate_tokens(text):
    cyrillic = sum(
        1 for c in text
        if '\u0400' <= c <= '\u04ff'
    )
    other = len(text) - cyrillic
    return max(
        1, cyrillic // 2 + other // 4
    )

class TokenTracker:
    def __init__(self):
        self.total = 0
        self.today = 0
        self.date = datetime.now().strftime(
            "%Y-%m-%d"
        )
        self.load()

    def load(self):
        try:
            if os.path.exists(TOKENS_FILE):
                with open(
                    TOKENS_FILE, "r",
                    encoding="utf-8"
                ) as f:
                    data = json.loads(
                        f.read().strip()
                    )
                self.total = data.get(
                    "total", 0
                )
                today = (
                    datetime.now().strftime(
                        "%Y-%m-%d"
                    )
                )
                if data.get("date") == today:
                    self.today = data.get(
                        "today", 0
                    )
                else:
                    self.today = 0
                    self.date = today
        except Exception:
            pass

    def add(self, input_text, output_text):
        input_tok = estimate_tokens(
            input_text
        )
        output_tok = estimate_tokens(
            output_text
        )
        total = input_tok + output_tok
        self.total += total
        self.today += total
        self.save()
        log(
            "TOKEN",
            f"in:{input_tok} "
            f"out:{output_tok} "
            f"sum:{total} "
            f"today:{self.today} "
            f"all:{self.total}"
        )
        return total

    def save(self):
        try:
            data = {
                "total": self.total,
                "today": self.today,
                "date": self.date
            }
            with open(
                TOKENS_FILE, "w",
                encoding="utf-8"
            ) as f:
                f.write(json.dumps(data))
        except Exception:
            pass

    def get_display(self):
        if self.total < 1000:
            total_str = str(self.total)
        elif self.total < 1000000:
            total_str = (
                f"{self.total // 1000}K"
            )
        else:
            total_str = (
                f"{self.total / 1000000:.1f}M"
            )
        if self.today < 1000:
            today_str = str(self.today)
        else:
            today_str = (
                f"{self.today // 1000}K"
            )
        return (
            f"Tokens: {today_str} today"
            f" / {total_str} total"
        )

def get_weather():
    now = time.time()
    if weather_cache["data"]:
        if now - weather_cache["time"] < 1800:
            return weather_cache["data"]
    try:
        url = (
            "http://api.openweathermap.org"
            "/data/2.5/weather"
            f"?q={WEATHER_CITY}"
            f"&appid={WEATHER_KEY}"
            "&units=metric&lang=ru"
        )
        resp = requests.get(url, timeout=5)
        data = resp.json()
        temp = round(data["main"]["temp"])
        feels = round(
            data["main"]["feels_like"]
        )
        desc = data["weather"][0][
            "description"
        ]
        humid = data["main"]["humidity"]
        wind = round(data["wind"]["speed"])
        result = (
            f"Weather: {temp}C "
            f"(feels {feels}C), "
            f"{desc}, humidity {humid}%, "
            f"wind {wind} m/s. "
        )
        if temp < -10:
            result += "Very cold! "
        elif temp < 0:
            result += "Cold. "
        elif temp < 10:
            result += "Cool. "
        elif temp < 20:
            result += "Nice. "
        elif temp < 30:
            result += "Warm. "
        else:
            result += "Hot! "
        weather_cache["data"] = result
        weather_cache["time"] = now
        return result
    except Exception as e:
        log("ERROR", f"Weather error: {e}")
        return "Weather unknown. "

def get_weather_condition():
    data = weather_cache.get("data", "")
    if not data:
        return "unknown"
    d = data.lower()
    if any(w in d for w in [
        "дожд", "rain", "ливен"
    ]):
        return "rain"
    if any(w in d for w in [
        "снег", "snow", "метел"
    ]):
        return "snow"
    if any(w in d for w in [
        "облач", "cloud", "пасмурн",
        "overcast"
    ]):
        return "cloudy"
    if any(w in d for w in [
        "ясно", "clear", "солн"
    ]):
        return "sunny"
    if "very cold" in d:
        return "freezing"
    if "cold" in d:
        return "cold"
    if "hot" in d:
        return "hot"
    return "normal"

def safe_load(fp):
    if os.path.exists(fp):
        try:
            with open(
                fp, "r", encoding="utf-8"
            ) as f:
                t = f.read().strip()
            if t:
                return json.loads(t)
        except Exception:
            bak = fp + ".bak"
            if os.path.exists(bak):
                try:
                    with open(
                        bak, "r",
                        encoding="utf-8"
                    ) as f:
                        t = f.read().strip()
                    if t:
                        return json.loads(t)
                except Exception:
                    pass
    return None

def safe_save(fp, data):
    try:
        t = json.dumps(
            data, ensure_ascii=False,
            indent=2
        )
        tmp = fp + ".tmp"
        bak = fp + ".bak"
        with open(
            tmp, "w", encoding="utf-8"
        ) as f:
            f.write(t)
        with open(
            tmp, "r", encoding="utf-8"
        ) as f:
            json.loads(f.read())
        if os.path.exists(fp):
            if os.path.exists(bak):
                os.remove(bak)
            os.rename(fp, bak)
        os.rename(tmp, fp)
    except Exception as e:
        log(
            "ERROR",
            f"Save error {fp}: {e}"
        )
        bak = fp + ".bak"
        if (
            os.path.exists(bak)
            and not os.path.exists(fp)
        ):
            os.rename(bak, fp)

def save_settings_file():
    safe_save(SETTINGS_FILE, user_settings)

class ClosenessSystem:
    def __init__(self):
        self.points = 0
        self.level = 1
        self.total_messages = 0
        self.positive_count = 0
        self.negative_count = 0
        self.first_met = (
            datetime.now().isoformat()
        )
        self.last_interaction = None
        self.daily_streak = 0
        self.last_streak_date = None
        self.milestones = []
        self.session_messages = 0
        self._lock = threading.Lock()
        self.load()

    def load(self):
        data = safe_load(CLOSENESS_FILE)
        if data and isinstance(data, dict):
            self.points = data.get(
                "points", 0
            )
            self.level = data.get(
                "level", 1
            )
            self.total_messages = data.get(
                "total_messages", 0
            )
            self.positive_count = data.get(
                "positive_count", 0
            )
            self.negative_count = data.get(
                "negative_count", 0
            )
            self.first_met = data.get(
                "first_met",
                datetime.now().isoformat()
            )
            self.last_interaction = (
                data.get("last_interaction")
            )
            self.daily_streak = data.get(
                "daily_streak", 0
            )
            self.last_streak_date = data.get(
                "last_streak_date"
            )
            self.milestones = data.get(
                "milestones", []
            )
            log(
                "INFO",
                f"Closeness loaded: "
                f"L{self.level} "
                f"({self.points}pts)"
            )

    def save(self):
        data = {
            "points": self.points,
            "level": self.level,
            "total_messages":
                self.total_messages,
            "positive_count":
                self.positive_count,
            "negative_count":
                self.negative_count,
            "first_met": self.first_met,
            "last_interaction":
                self.last_interaction,
            "daily_streak":
                self.daily_streak,
            "last_streak_date":
                self.last_streak_date,
            "milestones": self.milestones,
        }
        safe_save(CLOSENESS_FILE, data)

    def on_message(
        self, user_emotion="neutral"
    ):
        with self._lock:
            self.total_messages += 1
            self.session_messages += 1
            self.points += 1
            if user_emotion in (
                "happy", "excited", "tender"
            ):
                self.points += 2
                self.positive_count += 1
            elif user_emotion in ("angry",):
                self.points = max(
                    0, self.points - 1
                )
                self.negative_count += 1
            if self.session_messages == 5:
                self.points += 3
                self._add_milestone(
                    "first_long_chat"
                )
            elif self.session_messages == 15:
                self.points += 5
                self._add_milestone(
                    "marathon_chat"
                )
            elif self.session_messages == 50:
                self.points += 10
                self._add_milestone(
                    "epic_conversation"
                )
            self._update_streak()
            milestones_msgs = {
                10: "10_messages",
                50: "50_messages",
                100: "100_messages",
                250: "250_messages",
                500: "500_messages",
                1000: "1000_messages",
            }
            if (
                self.total_messages
                in milestones_msgs
            ):
                m = milestones_msgs[
                    self.total_messages
                ]
                self._add_milestone(m)
                self.points += 5
            old_level = self.level
            self._update_level()
            self.last_interaction = (
                datetime.now().isoformat()
            )
            self.save()
            if self.level > old_level:
                log(
                    "CLOSE",
                    f"LEVEL UP! "
                    f"{old_level} -> "
                    f"{self.level} "
                    f"({self.points}pts)"
                )
                return self.level
            return None

    def on_absence(self, hours):
        with self._lock:
            if hours > 72:
                penalty = min(
                    20,
                    int(hours / 24) * 3
                )
                self.points = max(
                    0, self.points - penalty
                )
                self._update_level()
                self.save()
                log(
                    "CLOSE",
                    f"Absence penalty: "
                    f"-{penalty}pts "
                    f"({hours:.0f}h away)"
                )
            elif hours > 24:
                penalty = min(
                    5, int(hours / 24)
                )
                self.points = max(
                    0, self.points - penalty
                )
                self.save()

    def on_personal_share(self):
        with self._lock:
            self.points += 3
            self._add_milestone(
                "personal_share"
            )
            self._update_level()
            self.save()

    def _update_streak(self):
        today = datetime.now().strftime(
            "%Y-%m-%d"
        )
        if self.last_streak_date:
            yesterday = (
                datetime.now()
                - timedelta(days=1)
            ).strftime("%Y-%m-%d")
            if (
                self.last_streak_date
                == yesterday
            ):
                self.daily_streak += 1
                bonus = min(
                    5, self.daily_streak
                )
                self.points += bonus
                streak_milestones = {
                    3: "streak_3",
                    7: "streak_week",
                    14: "streak_2weeks",
                    30: "streak_month",
                }
                if (
                    self.daily_streak
                    in streak_milestones
                ):
                    m = streak_milestones[
                        self.daily_streak
                    ]
                    self._add_milestone(m)
                    self.points += 10
            elif (
                self.last_streak_date != today
            ):
                self.daily_streak = 1
        else:
            self.daily_streak = 1
        self.last_streak_date = today

    def _update_level(self):
        for lvl in sorted(
            CLOSENESS_LEVELS.keys(),
            reverse=True
        ):
            if self.points >= (
                CLOSENESS_LEVELS[lvl][
                    "min_points"
                ]
            ):
                self.level = lvl
                break

    def _add_milestone(self, name):
        if name not in self.milestones:
            self.milestones.append(name)
            log(
                "CLOSE",
                f"Milestone: {name}"
            )

class MemoryDB:
    def __init__(self):
        self._ready = False
        self._client = None
        self._collection = None
        self._msg_counter = 0
        self._summaries = []
        self._structured_facts = {}
        self._init_lock = threading.Lock()
        self._init_db()
        self._load_summaries()
        self._load_structured_facts()

    def _init_db(self):
        try:
            if chromadb is None:
                log(
                    "WARN",
                    "ChromaDB not installed"
                )
                return
            os.makedirs(
                VECTOR_DB_PATH,
                exist_ok=True
            )
            self._client = (
                chromadb.PersistentClient(
                    path=VECTOR_DB_PATH
                )
            )
            self._collection = (
                self._client
                .get_or_create_collection(
                    name="conversations",
                    metadata={
                        "hnsw:space": "cosine"
                    }
                )
            )
            self._msg_counter = (
                self._collection.count()
            )
            self._ready = True
            log(
                "MEMORY",
                f"ChromaDB ready: "
                f"{self._msg_counter} "
                f"messages stored"
            )
        except Exception as e:
            log(
                "ERROR",
                f"ChromaDB init failed: {e}"
            )
            self._ready = False

    def _get_embeddings(self, texts):
        try:
            with GigaChat(
                credentials=GIGACHAT_KEY,
                verify_ssl_certs=False,
                model="GigaChat"
            ) as giga:
                result = giga.embeddings(
                    texts=texts,
                    model="Embeddings"
                )
                return [
                    item.embedding
                    for item in result.data
                ]
        except Exception as e:
            log(
                "ERROR",
                f"Embeddings error: {e}"
            )
            return None

    def store_message(
        self, role, content,
        emotion=""
    ):
        if not self._ready:
            return
        try:
            clean = re.sub(
                r'\[.*?\]', '', content
            ).strip()
            if len(clean) < 5:
                return
            embeddings = (
                self._get_embeddings(
                    [clean]
                )
            )
            if not embeddings:
                return
            self._msg_counter += 1
            msg_id = (
                f"msg_{int(time.time())}"
                f"_{self._msg_counter}"
            )
            timestamp = (
                datetime.now().isoformat()
            )
            who = (
                "User" if role == "user"
                else "Yulya"
            )
            self._collection.add(
                documents=[
                    f"{who}: {clean}"
                ],
                embeddings=embeddings,
                metadatas=[{
                    "role": role,
                    "emotion": emotion,
                    "timestamp": timestamp,
                    "length": len(clean),
                }],
                ids=[msg_id]
            )
            log(
                "MEMORY",
                f"Stored: {who}: "
                f"{clean[:50]}..."
            )
        except Exception as e:
            log(
                "ERROR",
                f"Store message error: {e}"
            )

    def search_relevant(
        self, query, n_results=None
    ):
        if not self._ready:
            return []
        if n_results is None:
            n_results = MEMORY_SEARCH_RESULTS
        try:
            embeddings = (
                self._get_embeddings(
                    [query]
                )
            )
            if not embeddings:
                return []
            total = self._collection.count()
            if total == 0:
                return []
            actual_n = min(
                n_results, total
            )
            results = (
                self._collection.query(
                    query_embeddings=(
                        embeddings
                    ),
                    n_results=actual_n
                )
            )
            found = []
            if results and (
                results['documents']
            ):
                docs = (
                    results['documents'][0]
                )
                metas = (
                    results['metadatas'][0]
                )
                dists = (
                    results['distances'][0]
                    if 'distances' in results
                    else [0] * len(docs)
                )
                for doc, meta, dist in zip(
                    docs, metas, dists
                ):
                    if dist < 0.5:
                        found.append({
                            "text": doc,
                            "timestamp": (
                                meta.get(
                                    "timestamp",
                                    ""
                                )
                            ),
                            "role": (
                                meta.get(
                                    "role", ""
                                )
                            ),
                            "distance": dist,
                        })
            if found:
                log(
                    "MEMORY",
                    f"Found {len(found)} "
                    f"relevant for: "
                    f"{query[:40]}"
                )
            return found
        except Exception as e:
            log(
                "ERROR",
                f"Search error: {e}"
            )
            return []

    def build_memory_context(
        self, user_text
    ):
        parts = []
        relevant = self.search_relevant(
            user_text
        )
        if relevant:
            memories = []
            for r in relevant:
                ts = r.get("timestamp", "")
                try:
                    dt = (
                        datetime.fromisoformat(
                            ts
                        )
                    )
                    ago = (
                        datetime.now() - dt
                    )
                    if ago.days > 0:
                        when = (
                            f"{ago.days}d ago"
                        )
                    else:
                        hours = (
                            ago.seconds // 3600
                        )
                        when = (
                            f"{hours}h ago"
                        )
                except Exception:
                    when = "earlier"
                memories.append(
                    f"[{when}] {r['text']}"
                )
            parts.append(
                "MEMORIES from past "
                "conversations "
                "(use naturally, don't "
                "list them):\n"
                + "\n".join(memories)
            )
        if self._summaries:
            recent = self._summaries[-3:]
            sum_text = "\n".join(
                f"- {s.get('summary', '')}"
                for s in recent
            )

                "RECENT SESSION "
                "SUMMARIES:\n" + sum_text
            )
        if self._structured_facts:
            facts_lines = []
            for cat, val in (

            ):

                        if val:
                            val_str = (
                                ", ".join(
                                    str(v)
                                    for v in val
                                )
                            )
                        else:
                            continue
                    else:
                        val_str = str(val)
                    facts_lines.append(
                        f" {cat}: {val_str}"
                    )
            if facts_lines:
                parts.append(
                    "USER PROFILE:\n"
                    + "\n".join(facts_lines)
                )
        return "\n".join(parts)

def load_mood():
    if os.path.exists(MOOD_FILE):
        try:
            with open(
                MOOD_FILE, "r",
                encoding="utf-8"
            ) as f:
                data = json.loads(
                    f.read().strip()
                )
            st = data.get("time", 0)
            sm = data.get("mood", 0)
            hours = (
                (time.time() - st) / 3600
            )
            if hours > 8:
                return 0
            elif hours > 4:
                decay = 2
            elif hours > 2:
                decay = 1
            else:
                return sm
            if sm > 0:
                return max(0, sm - decay)
            elif sm < 0:
                return min(0, sm + decay)
            return 0
        except Exception:
            return 0
    return 0

def get_life_events(hours_away):
    events = []
    periods_count = max(
        1, min(3, int(hours_away / 4))
    )
    now_hour = datetime.now().hour
    if 5 <= now_hour < 12:
        periods = [
            "morning", "night", "evening"
        ]
    elif 12 <= now_hour < 17:
        periods = [
            "morning", "afternoon",
            "morning"
        ]
    elif 17 <= now_hour < 22:
        periods = [
            "afternoon", "morning",
            "afternoon"
        ]
    else:
        periods = [
            "evening", "afternoon",
            "evening"
        ]
    weather_cond = get_weather_condition()
    weather_pool = WEATHER_LIFE_EVENTS.get(
        weather_cond, []
    )
    for i in range(periods_count):
        p = periods[i % len(periods)]
        pool = LIFE_EVENTS.get(
            p, LIFE_EVENTS["afternoon"]
        )
        if (
            weather_pool
            and random.random() < 0.4
        ):
            event = random.choice(
                weather_pool
            )
        else:
            event = random.choice(pool)
        if event not in events:
            events.append(event)
        else:
            event = random.choice(pool)
            events.append(event)
    return events

def build_absence_context():
    hours = get_absence_hours()
    if hours < 0:
        return (
            "First time meeting user. "
            "Greet warmly."
        )
    if hours < 0.5:
        category = "short"
    elif hours < 2:
        category = "medium"
    elif hours < 4:
        category = "long"
    elif hours < 8:
        category = "very_long"
    elif hours < 24:
        category = "day"
    elif hours < 168:
        category = "days"
    else:
        category = "week"
    events = get_life_events(hours)
    template = random.choice(
        ABSENCE_REACTIONS[category]
    )
    days_away = int(hours / 24)
    if days_away < 1:
        days_away = 1
    result = template
    if "{event}" in result and events:
        result = result.replace(

        )
    if (
        "{event2}" in result

        result = result.replace(
            "{event2}", events[1]

    if "{days}" in result:
        result = result.replace(

        )
    hours_int = int(hours)
    info = closeness.get_level_info()
    absence_mult = info.get(
        "absence_mult", 1.0
    )
    uname = user_settings.get(

def cleanup_old_events():
    now = datetime.now()
    kept = []
    for ev in calendar_events:
        d = ev.get("date", "")
        etype = ev.get("type", "event")
        if etype == "birthday":
            kept.append(ev)
            continue
        try:
            if len(d) == 10:
                ev_date = datetime.strptime(
                    d, "%Y-%m-%d"
                )
                if (now - ev_date).days <= 7:
                    kept.append(ev)
            else:
                kept.append(ev)
        except ValueError:
            kept.append(ev)
    removed = (
        len(calendar_events) - len(kept)
    )
    calendar_events.clear()
    calendar_events.extend(kept)
    if removed > 0:
        save_events()
        log(
            "EVENT",
            f"Cleaned {removed} old events"
        )

def check_follow_ups():
    yesterday = (
        datetime.now() - timedelta(days=1)
    )
    y_md = yesterday.strftime("%m-%d")
    y_full = yesterday.strftime("%Y-%m-%d")
    found = []
    for ev in calendar_events:
        d = ev.get("date", "")
        fu = ev.get("follow_up", "")
        done = ev.get("followed", False)
        if fu and not done:
            if d == y_md or d == y_full:
                found.append(ev)
    return found

def check_upcoming_events(days_ahead=3):
    now = datetime.now()
    found = []
    for i in range(1, days_ahead + 1):
        future = now + timedelta(days=i)
        f_md = future.strftime("%m-%d")
        f_full = future.strftime("%Y-%m-%d")
        for ev in calendar_events:
            d = ev.get("date", "")
            if d == f_md or d == f_full:
                ev_copy = ev.copy()
                ev_copy["days_until"] = i
                found.append(ev_copy)
    return found

def detect_event_in_text(text):
    t = text.lower()
    now = datetime.now()
    meaningful_words = [
        "день рождения", "день рожденья",
        "днюха", "встреча",
        "собеседование", "экзамен",
        "дедлайн", "праздник",
        "годовщина", "свадьба", "концерт",
        "врач", "доктор", "запись",
    ]
    date_words = [
        "завтра", "послезавтра", "через",
    ]
    has_meaningful = any(
        m in t for m in meaningful_words
    )
    has_date_word = any(
        d in t for d in date_words
    )
    has_month = False
    for month_key in MONTHS_RU:
        if month_key in t:
            has_month = True
            break
    if not has_meaningful:
        return None
    if not has_date_word and not has_month:
        return None
    target_date = None
    event_text = text
    if "послезавтра" in t:
        target_date = (
            now + timedelta(days=2)
        )
    elif "завтра" in t:
        target_date = (
            now + timedelta(days=1)
        )
    elif "через неделю" in t:
        target_date = (
            now + timedelta(days=7)
        )
    elif "через месяц" in t:
        target_date = (
            now + timedelta(days=30)
        )
    else:
        day_match = re.search(
            r'(\d{1,2})', t
        )
        month_found = None
        for mk, mn in MONTHS_RU.items():
            if mk in t:
                month_found = mn
                break
        if day_match and month_found:
            day = int(day_match.group(1))
            if 1 <= day <= 31:
                year = now.year
                try:
                    target_date = datetime(
                        year, month_found, day
                    )
                    if target_date < now:
                        target_date = (
                            datetime(
                                year + 1,
                                month_found,
                                day
                            )
                        )
                except ValueError:
                    return None
    if target_date is None:
        return None
    is_birthday = any(
        w in t for w in [
            "день рождения",
            "день рожденья", "днюха"
        ]
    )
    follow_up = ""
    event_words = [
        "собеседование", "экзамен",
        "встреча", "врач", "доктор",
        "концерт",
    ]
    for ew in event_words:
        if ew in t:
            follow_up = (
                f"Kak proshlo: {ew}?"
            )
            break
    if is_birthday:
        date_str = target_date.strftime(
            "%m-%d"
        )
        etype = "birthday"
    else:
        date_str = target_date.strftime(
            "%Y-%m-%d"
        )
        etype = "event"
    return {
        "date": date_str,
        "type": etype,
        "text": event_text[:100],
        "follow_up": follow_up,
        "followed": False,
    }

def parse_emotion(text):
    match = re.match(
        r'\[(\w+)\]\s*(.*)', text,
        re.DOTALL
    )
    if match:
        e = match.group(1).lower()
        t = match.group(2).strip()
        if e in EMOTIONS_VOICE:
            return e, t
        return guess_emotion(t), t
    text = re.sub(
        r'\[.*?\]', '', text
    ).strip()
    return guess_emotion(text), text

def clean_speech(text):
    text = re.sub(
        "[\U0001F600-\U0001F64F"
        "\U0001F300-\U0001F5FF"
        "\U0001F680-\U0001F6FF"
        "\U0001F1E0-\U0001F1FF"
        "\U00002702-\U000027B0"
        "\U000024C2-\U0001F251"
        "\U0001F900-\U0001F9FF"
        "\U0001FA00-\U0001FA6F"
        "\U00002600-\U000026FF"
        "\U00002700-\U000027BF"
        "\U0000FE00-\U0000FE0F"
        "\U0000200D]+",
        "", text
    )
    text = re.sub(
        r'[:;]-?[)(DPp\]\[/\\]', '', text
    )
    text = re.sub(r'\*[^*]+\*', '', text)
    text = re.sub(r'\s+', ' ', text).strip()
    return text

def add_speech_pauses(text):
    text = re.sub(r'\.{2,}', ', ,', text)
    text = re.sub(r'\s*[—–]\s*', ', ', text)
    fillers = [
        'ну', 'вот', 'короче', 'типа',
        'значит', 'слушай', 'знаешь',
        'кстати', 'допустим', 'впрочем',
        'собственно',
    ]
    for filler in fillers:
        pattern = (
            rf'(?<![а-яА-ЯёЁ])'
            rf'({filler})'
            rf'(?![а-яА-ЯёЁ,.:;!?])'
        )
        text = re.sub(
            pattern, r'\1,',
            text, flags=re.IGNORECASE
        )
    sounds = [
        'хмм', 'хм', 'ммм', 'мм', 'эм',
        'ээ', 'ааа', 'эхх', 'эх', 'ох',
        'ах', 'ой', 'уф', 'фух', 'пф',
    ]
    for sound in sounds:
        pattern = (
            rf'(?<![а-яА-ЯёЁ])({sound})'
            rf'(?![а-яА-ЯёЁ,.:;!?])'
        )
        text = re.sub(
            pattern, r'\1,',
            text, flags=re.IGNORECASE
        )
    laughs = [
        'хаха', 'хахаха', 'хех', 'хехе',
        'хихи', 'ахах', 'ахаха',
    ]
    for laugh in laughs:
        pattern = (
            rf'({laugh})'
            rf'(?![а-яА-ЯёЁ,.:;!?])'
        )
        text = re.sub(
            pattern, r'\1,',
            text, flags=re.IGNORECASE
        )
    reactions = [
        'блин', 'блиин', 'ого', 'вау',
        'ничего себе', 'да ладно',
        'серьёзно', 'круто', 'класс',
        'жесть', 'капец', 'офигеть',
        'обалдеть',
    ]
    for reaction in reactions:
        pattern = (
            rf'(?<![а-яА-ЯёЁ])({reaction})'
            rf'(?![а-яА-ЯёЁ,.:;!?])'
        )
        text = re.sub(
            pattern, r'\1,',
            text, flags=re.IGNORECASE
        )
    for conj in [
        'но', 'а', 'однако', 'зато',
        'хотя'
    ]:
        pattern = (
            rf'([а-яА-ЯёЁ])\s+({conj})\s'
        )
        text = re.sub(
            pattern, rf'\1, \2 ',
            text, flags=re.IGNORECASE
        )
    for name in [
        'дим', 'дима', 'дмитрий'
    ]:
        pattern = (
            rf'(?<![а-яА-ЯёЁ])({name})'
            rf'(?![а-яА-ЯёЁ,.:;!?])'
        )
        text = re.sub(
            pattern, r'\1,',
            text, flags=re.IGNORECASE
        )
    introductory = [

        'кажется', 'по-моему',
        'честно говоря', 'на самом деле',

    )
    text = re.sub(
        r',\s*,\s*,', ', ,', text
    )

def split_sentences(text):
    parts = re.split(
        r'(?<=[.!?])\s+', text
    )
    result = []
    for part in parts:
        part = part.strip()
        if part:
            result.append(part)
    if not result and text.strip():
        result.append(text.strip())
    return result

def text_to_speech_chunks(text):
    sentences = split_sentences(text)
    chunks = []
    current = ""
    for sent in sentences:
        if not current:
            current = sent
        elif (
            len(current) + len(sent)
        ) < 120:
            current += " " + sent
        else:
            chunks.append(current)
            current = sent
    if current:
        chunks.append(current)
    if not chunks and text.strip():
        chunks.append(text.strip())
    return chunks

def get_music_dir():
    return user_settings.get(
        "music_dir", MUSIC_DIR
    )

def scan_music_library():
    """
    Сканирует папку с музыкой.
    Возвращает словарь:
    { "genre_name": [filepath, ...] }
    """
    library = {}
    mdir = get_music_dir()
    if not os.path.exists(mdir):
        log(
            "MUSIC",
            f"Music dir not found: {mdir}"
        )
        return library
    # Сначала ищем по подпапкам (жанры)
    for item in os.listdir(mdir):
        item_path = os.path.join(mdir, item)
        if os.path.isdir(item_path):
            genre_key = item.lower().strip()
            files = []
            for fname in os.listdir(
                item_path
            ):
                if fname.lower().endswith(
                    MUSIC_EXTENSIONS
                ):
                    files.append(
                        os.path.join(
                            item_path, fname
                        )
                    )
            if files:
                library[genre_key] = files
                log(
                    "MUSIC",
                    f"Genre '{genre_key}': "
                    f"{len(files)} tracks"
                )
    # Если нет подпапок — корень как "all"
    if not library:
        files = []
        for fname in os.listdir(mdir):
            if fname.lower().endswith(
                MUSIC_EXTENSIONS
            ):
                files.append(
                    os.path.join(mdir, fname)
                )
        if files:
            library["all"] = files
            log(
                "MUSIC",
                f"Root music: "
                f"{len(files)} tracks"
            )
    return library

def find_genre_in_text(text):
    """
    Определяет жанр музыки из текста.
    Сначала проверяет реальные папки,
    потом словарь синонимов.
    """
    t = text.lower()

    # Шаг 1: смотрим реальные папки
    library = scan_music_library()
    # Сортируем по длине названия —
    # длинные совпадения важнее
    folders = sorted(
        library.keys(),
        key=len,
        reverse=True
    )
    for folder in folders:
        # Точное вхождение папки в текст
        if folder in t:
            log(
                "MUSIC",
                f"Genre by folder: {folder}"
            )
            return folder

    # Шаг 2: проверяем словарь синонимов
    # Сортируем по длине алиаса —
    # длинные важнее (нью эйдж > эйдж)
    matches = []
    for genre_key, aliases in (
        MUSIC_GENRES.items()
    ):
        for alias in aliases:
            if alias in t:
                matches.append(
                    (len(alias), genre_key)
                )

    if matches:
        # Берём самое длинное совпадение
        matches.sort(reverse=True)
        best_genre = matches[0][1]
        log(
            "MUSIC",
            f"Genre by alias: {best_genre}"
        )
        # Проверяем есть ли такая папка
        if best_genre in library:
            return best_genre
        # Ищем папку по частичному совпадению
        for folder in folders:
            if (
                best_genre in folder
                or folder in best_genre
            ):
                return folder
        # Возвращаем жанр даже без папки
        return best_genre

    # Шаг 3: если ничего не нашли
    # и есть только одна папка — берём её
    if len(library) == 1:
        only = list(library.keys())[0]
        return only

    log("MUSIC", "Genre not detected")
    return None

def detect_music_command(text):
    """
    Определяет команду управления музыкой.
    Возвращает (команда, жанр) или None.
    """
    t = text.lower().strip()

    # Стоп — проверяем первым
    for cmd in MUSIC_CMD_STOP:
        if cmd in t:
            return "stop", None

    # Пауза
    for cmd in MUSIC_CMD_PAUSE:
        if cmd in t:
            return "pause", None

    # Продолжить
    for cmd in MUSIC_CMD_RESUME:
        if cmd in t:
            return "resume", None

    # Следующая
    for cmd in MUSIC_CMD_NEXT:
        if cmd in t:
            return "next", None

    # Громче
    for cmd in MUSIC_CMD_LOUDER:
        if cmd in t:
            return "louder", None

    # Тише
    for cmd in MUSIC_CMD_QUIETER:
        if cmd in t:
            return "quieter", None

    # Включить — проверяем есть ли
    # команда включения
    has_play_cmd = any(
        cmd in t for cmd in MUSIC_CMD_PLAY
    )
    # Также проверяем просто "музыку"
    has_music_word = any(
        w in t for w in [
            "музык", "трек", "песн",
            "music", "play"
        ]
    )

    if has_play_cmd or has_music_word:
        genre = find_genre_in_text(t)
        return "play", genre

    return None, None

def music_play_genre(genre=None):
    """
    Запускает музыку указанного жанра.
    Если жанр None — случайный.
    """
    global music_state
    library = scan_music_library()
    if not library:
        log("MUSIC", "Library empty")
        return False, "Папка с музыкой пуста"

    # Ищем папку жанра
    target_files = None
    target_genre = genre

    if genre:
        # Точное совпадение
        if genre in library:
            target_files = library[genre]
        else:
            # Частичное совпадение
            for folder, files in (
                library.items()
            ):
                if (
                    genre in folder
                    or folder in genre
                ):
                    target_files = files
                    target_genre = folder
                    break
            # Через словарь синонимов
            if target_files is None:
                for gkey, aliases in (
                    MUSIC_GENRES.items()
                ):
                    if genre in aliases:
                        if gkey in library:
                            target_files = (
                                library[gkey]
                            )
                            target_genre = gkey
                            break

    # Если жанр не найден — случайный
    if target_files is None:
        target_genre = random.choice(
            list(library.keys())
        )
        target_files = library[target_genre]

    if not target_files:
        return False, "Треки не найдены"

    # Перемешиваем плейлист
    playlist = target_files.copy()
    random.shuffle(playlist)

    music_state["playlist"] = playlist
    music_state["playlist_index"] = 0
    music_state["current_genre"] = (
        target_genre
    )

    return _music_start_track(0), target_genre

def _music_start_track(index):
    """Запускает трек по индексу."""
    global music_state
    playlist = music_state["playlist"]
    if not playlist:
        return False
    if index >= len(playlist):
        index = 0
        music_state["playlist_index"] = 0

    filepath = playlist[index]
    music_state["current_file"] = filepath
    music_state["playlist_index"] = index

    try:
        # Останавливаем текущее
        try:
            pygame.mixer.music.stop()
            pygame.mixer.music.unload()
        except Exception:
            pass

        pygame.mixer.music.load(filepath)
        vol = music_state.get("volume", 0.5)
        pygame.mixer.music.set_volume(vol)
        pygame.mixer.music.play()
        music_state["playing"] = True
        music_state["paused"] = False

        fname = os.path.basename(filepath)
        log(
            "MUSIC",
            f"Playing: {fname}"
        )
        return True
    except Exception as e:
        log(
            "ERROR",
            f"Music play error: {e}"
        )
        return False

def music_stop():
    """Останавливает музыку."""
    global music_state
    try:
        pygame.mixer.music.stop()
        pygame.mixer.music.unload()
    except Exception:
        pass
    music_state["playing"] = False
    music_state["paused"] = False
    music_state["current_file"] = None
    music_state["current_genre"] = None
    music_state["playlist"] = []
    log("MUSIC", "Stopped")

def music_pause():
    """Ставит на паузу."""
    global music_state
    try:
        if music_state["playing"]:
            pygame.mixer.music.pause()
            music_state["playing"] = False
            music_state["paused"] = True
            log("MUSIC", "Paused")
            return True
    except Exception as e:
        log("ERROR", f"Pause error: {e}")
    return False

def music_resume():
    """Снимает с паузы."""
    global music_state
    try:
        if music_state["paused"]:
            pygame.mixer.music.unpause()
            music_state["playing"] = True
            music_state["paused"] = False
            log("MUSIC", "Resumed")
            return True
    except Exception as e:
        log("ERROR", f"Resume error: {e}")
    return False

def music_next():
    """Следующий трек."""
    global music_state
    playlist = music_state["playlist"]
    if not playlist:
        return False
    idx = music_state["playlist_index"] + 1
    if idx >= len(playlist):
        idx = 0
    return _music_start_track(idx)

def music_set_volume(delta):
    """
    Меняет громкость.
    delta: +0.1 громче, -0.1 тише
    """
    global music_state
    vol = music_state.get("volume", 0.5)
    vol = max(0.0, min(1.0, vol + delta))
    music_state["volume"] = vol
    user_settings["music_volume"] = vol
    try:
        pygame.mixer.music.set_volume(vol)
    except Exception:
        pass
    log("MUSIC", f"Volume: {vol:.1f}")
    return vol

def music_monitor_thread():
    """
    Фоновый поток — следит за концом трека
    и запускает следующий.
    """
    while True:
        time.sleep(2)
        try:
            if (
                music_state["playing"]
                and not music_state["paused"]
                and not pygame.mixer.music
                .get_busy()
            ):
                # Трек закончился
                log(
                    "MUSIC",
                    "Track ended, next..."
                )
                music_next()
        except Exception:
            pass

                f"{genre_display} — хороший выбор!",
                f"Сейчас поставлю {genre_display}",

        else:

                f"Не нашла музыку, {uname}. "
                f"Проверь папку "
                f"{get_music_dir()}"

    elif cmd == "stop":

            "Выключила музыку",
            "Ок, тишина",

        ]

    elif cmd == "resume":
        ok = music_resume()
        if ok:
            return "Продолжаю!"
        return "Нечего продолжать"
    elif cmd == "next":
        ok = music_next()

def detect_help_needed(text):
    """
    Определяет нужна ли помощь
    по триггерным словам трудности.
    """
    t = text.lower()

        1 for w in HELP_TRIGGER_WORDS
        if w in t
    )

def detect_document_request(text):
    t = text.lower()
    pdf_words = [
        "в pdf", "в пдф", "pdf",
        "пдф", "пдф файл", "pdf файл",
    ]
    html_words = [
        "в html", "в хтмл", "html",
        "хтмл", "в браузере",
    ]
    doc_words = [
        "сохрани", "запиши",
        "оформи", "подготовь",
        "сделай файл", "создай файл",
        "приготовь", "составь",
        "напиши инструкцию",
        "сделай инструкцию",
        "гайд", "мануал",
    ]
    is_pdf = any(
        w in t for w in pdf_words
    )
    is_html = any(
        w in t for w in html_words
    )
    is_doc = any(
        w in t for w in doc_words
    )
    if is_pdf or is_html:
        doc_type = (
            "pdf" if is_pdf else "html"
        )
        topic = detect_document_topic(
            text
        )
        return doc_type, topic
    content_words = [
        "инструкци", "рецепт",
        "файл", "гайд", "мануал",
        "список", "план", "шаги",
    ]
    has_content = any(
        w in t for w in content_words
    )
    if is_doc and has_content:
        topic = detect_document_topic(
            text
        )
        return "html", topic
    return None, None


def detect_document_topic(text):
    t = text.lower()
    known = {
        "python": [
            "python", "питон",
            "пайтон", "pip",
            "venv", "программ",
        ],
        "windows": [
            "windows", "виндовс",
            "винда", "реестр",
            "компьютер", "драйвер",
            "видеодрайвер", "видеокарт",
            "nvidia", "amd", "intel",
            "диспетчер",
            "безопасный режим",
        ],
        "figma": [
            "figma", "фигма",
        ],
        "git": [
            "git", "гит",
            "github", "гитхаб",
        ],
        "wifi": [
            "wifi", "вай-фай",
            "интернет", "роутер",
            "сеть",
        ],
    }
    for topic_key, keywords in (
        known.items()
    ):
        if any(
            kw in t for kw in keywords
        ):
            return topic_key
    stop_words = {
        "в", "на", "и", "или", "для",
        "по", "из", "от", "до", "за",
        "pdf", "html", "пдф", "файл",
        "сделай", "подготовь", "оформи",
        "мне", "пожалуйста",
        "можешь", "можно",
    }
    words = t.split()
    topic_words = [
        w for w in words
        if w not in stop_words
        and len(w) > 3
    ]
    if topic_words:
        return " ".join(
            topic_words[:3]
        )
    return "general"

def _text_to_html_body(text):
    """
    Конвертирует текст в HTML.
    Обрабатывает ### заголовки,
    - списки, **жирный**, нумерацию.
    """
    lines = text.split('\n')
    html_parts = []
    in_list = False
    in_ol = False

    for line in lines:
        line = line.strip()
        if not line:
            if in_list:
                html_parts.append("</ul>")
                in_list = False
            if in_ol:
                html_parts.append("</ol>")
                in_ol = False
            continue

        # ### Заголовок 3
        if line.startswith("### "):
            t = line[4:].strip()
            t = _inline_format(t)
            html_parts.append(
                f"<h3>{t}</h3>"
            )
        # ## Заголовок 2
        elif line.startswith("## "):
            t = line[3:].strip()
            t = _inline_format(t)
            html_parts.append(
                f"<h2>{t}</h2>"
            )
        # # Заголовок 1
        elif line.startswith("# "):
            t = line[2:].strip()
            t = _inline_format(t)
            html_parts.append(
                f"<h2>{t}</h2>"
            )
        # Нумерованный список
        elif re.match(
            r'^\d+[\.\)]\s', line
        ):
            if not in_ol:
                if in_list:
                    html_parts.append(
                        "</ul>"
                    )
                    in_list = False
                html_parts.append("<ol>")
                in_ol = True
            t = re.sub(
                r'^\d+[\.\)]\s*', '',
                line
            )
            t = _inline_format(t)
            html_parts.append(
                f"<li>{t}</li>"
            )
        # Маркированный список
        elif line.startswith(
            ("- ", "* ", "• ")
        ):
            if not in_list:
                if in_ol:
                    html_parts.append(
                        "</ol>"
                    )
                    in_ol = False
                html_parts.append("<ul>")
                in_list = True
            t = line[2:].strip()
            t = _inline_format(t)
            html_parts.append(
                f"<li>{t}</li>"
            )
        # Обычный параграф
        else:
            if in_list:
                html_parts.append("</ul>")
                in_list = False
            if in_ol:
                html_parts.append("</ol>")
                in_ol = False
            t = _inline_format(line)
            html_parts.append(
                f"<p>{t}</p>"
            )

    if in_list:
        html_parts.append("</ul>")
    if in_ol:
        html_parts.append("</ol>")

    return "\n".join(html_parts)

def _inline_format(text):
    """Форматирует **жирный** и `код`."""
    # **жирный**
    text = re.sub(
        r'\*\*(.+?)\*\*',
        r'<strong>\1</strong>',
        text
    )
    # *курсив*
    text = re.sub(
        r'\*(.+?)\*',
        r'<em>\1</em>',
        text
    )
    # `код`
    text = re.sub(
        r'`(.+?)`',
        r'<code>\1</code>',
        text
    )
    return text

def open_help_file(filepath):
    """Открывает HTML файл в браузере."""
    try:
        webbrowser.open(
            f"file:///{filepath}"
            .replace("\\", "/")
        )
        log(
            "HELP",
            f"Opened in browser: {filepath}"
        )
        return True
    except Exception as e:
        log(
            "ERROR",
            f"Open browser error: {e}"
        )
        return False

def is_duplicate_fact(new_fact, existing):
    new_words = set(
        new_fact.lower().split()
    )
    if len(new_words) == 0:
        return True
    for fact in existing:
        old_words = set(
            fact.lower().split()
        )
        overlap = len(
            new_words & old_words
        )
        bigger = max(
            len(new_words), len(old_words)
        )
        if bigger > 0:
            similarity = overlap / bigger
            if similarity > 0.6:
                return True
    return False

def trim_context_by_tokens(messages):
    sys_msgs = [
        m for m in messages
        if m["role"] == "system"
    ]
    chat_msgs = [
        m for m in messages
        if m["role"] != "system"
    ]
    budget = MAX_CONTEXT_TOKENS
    kept = []
    used = 0
    for m in reversed(chat_msgs):
        t = estimate_tokens(m["content"])
        if used + t > budget and kept:
            break
        kept.insert(0, m)
        used += t
    return sys_msgs + kept

def validate_response(reply):
    global last_response_text
    issues = []
    clean = re.sub(
        r'\[.*?\]', '', reply
    ).strip()
    latin_long = re.findall(
        r'[a-zA-Z]{10,}', clean
    )
    if len(latin_long) > 2:
        issues.append("too_much_english")
    ai_words = [
        "языковая модель", "я ии",
        "я бот", "нейросеть",
        "я программа",
        "искусственный интеллект",
        "я не могу иметь чувства",
        "как ии", "как бот",
        "i am an ai", "i'm an ai",
        "language model",
    ]
    lower = clean.lower()
    for w in ai_words:
        if w in lower:
            issues.append("broke_character")
            break
    if has_disclaimer(clean):
        issues.append("disclaimer")
    if len(clean) < 3:
        issues.append("too_short")
    if (
        last_response_text
        and clean == last_response_text
    ):
        issues.append("exact_repeat")
    last_response_text = clean
    if issues:
        log(
            "VALID",
            f"Issues: {issues} "
            f"in: {clean[:80]}"
        )
    return issues

def call_gigachat(
    messages, temp=0.8, max_tok=300
):
    retries = 3
    input_text = " ".join(
        m["content"] for m in messages
    )
    for attempt in range(retries):
        try:
            msgs = to_messages(messages)
            with GigaChat(
                credentials=GIGACHAT_KEY,
                verify_ssl_certs=False,
                model="GigaChat"
            ) as giga:
                resp = giga.chat(Chat(
                    messages=msgs,
                    temperature=temp,
                    max_tokens=max_tok
                ))
                reply = (
                    resp.choices[0]
                    .message.content
                )
                reply = (
                    strip_gigachat_disclaimer(
                        reply
                    )
                )
                if not reply or (
                    not reply.strip()
                ):
                    reply = (
                        "[neutral] Хм, что-то "
                        "я задумалась..."
                    )
                token_tracker.add(
                    input_text, reply
                )
                return reply
        except Exception as e:
            log(
                "ERROR",
                f"GigaChat attempt "
                f"{attempt+1}/{retries}: "
                f"{e}"
            )
            if attempt < retries - 1:
                time.sleep(2 ** attempt)
            else:
                raise e

            history_dicts.extend(
                chat_m[-MAX_MEMORY:]
            )

    save_memory()
    save_session()

    _, clean_reply = parse_emotion(reply)
    with pending_lock:
        if "?" in clean_reply:
            sentences = re.split(
                r'[.!]', clean_reply
            )
            questions = [
                s.strip()
                for s in sentences
                if "?" in s
            ]
            if questions:
                pending_question = (
                    questions[-1]
                )
            else:
                pending_question = (
                    clean_reply
                )
        else:
            pending_question = None

    msg_count += 1
    if msg_count % FACTS_INTERVAL == 0:
        threading.Thread(
            target=do_extract_facts,
            daemon=True
        ).start()

    threading.Thread(
        target=memory_db.store_message,
        args=("user", user_text),
        daemon=True
    ).start()
    em_tag, _ = parse_emotion(reply)
    threading.Thread(
        target=memory_db.store_message,
        args=("assistant", reply, em_tag),
        daemon=True
    ).start()

    if msg_count % FACTS_INTERVAL == 0:
        threading.Thread(
            target=(
                memory_db
                .update_structured_facts
            ),
            args=(history_dicts.copy(),),
            daemon=True
        ).start()

    return reply

        )

            INITIATIVE_PROMPTS + w_prompts
        )
        available = [
            p for p in all_prompts
            if p not in
            used_initiative_prompts
        ]
        if not available:
            used_initiative_prompts.clear()
            available = all_prompts[:]
        if (
            w_prompts
            and random.random() < 0.35
        ):
            w_avail = [
                p for p in w_prompts
                if p in available
            ]
            if w_avail:
                prompt = random.choice(
                    w_avail
                )
            else:
                prompt = random.choice(
                    available
                )
        else:
            gen_avail = [
                p for p in
                INITIATIVE_PROMPTS
                if p in available
            ]
            if gen_avail:
                prompt = random.choice(
                    gen_avail
                )
            else:
                prompt = random.choice(
                    available
                )
        used_initiative_prompts.add(prompt)

    events_ctx = build_events_context()
    if (
        events_ctx
        and random.random() < 0.4
    ):
        prompt = (
            "Mention upcoming event: "
            + events_ctx
            + " Speak Russian."
        )

    with history_lock:
        temp = history_dicts.copy()
        temp[0] = {
            "role": "system",
            "content": make_prompt()
        }
        temp.append({
            "role": "user",
            "content": prompt
        })

        temp, temp=0.9, max_tok=200
    )
    log("INIT", f"Initiative: {reply}")

    with history_lock:
        history_dicts.append({
            "role": "assistant",
            "content": reply
        })
    return reply

def do_extract_facts():

    with history_lock:

            m for m in history_dicts

                "user", "assistant"

        "Return JSON list of strings "
        "in Russian. "
        'Example: ["Zovut Dima",'
        '"Lyubit igry"] '
        "Conversation:\n" + conv
    )
    try:
        tmp = [
            {"role": "system",
             "content": "Extract facts."},
            {"role": "user",
             "content": prompt}
        ]

                if not is_duplicate_fact(

                    added += 1
            if added > 0:
                save_facts()
                log(
                    "FACTS",
                    f"Extracted {len(new)}, "
                    f"added {added} new"
                )
    except Exception as e:
        log("ERROR", f"Facts error: {e}")

# TTS — EDGE (облако)
# ==========================================
async def gen_speech_edge(
    text, emotion, out_file
):
    p = EMOTIONS_VOICE.get(
        emotion, EMOTIONS_VOICE["neutral"]
    )
    base_rate = p["rate"]
    rate_val = int(
        base_rate.replace("%", "")
        .replace("+", "")
    )
    rate_val += user_settings.get(
        "speech_rate_offset", 0
    )
    if rate_val >= 0:
        rate_str = f"+{rate_val}%"
    else:
        rate_str = f"{rate_val}%"
    comm = edge_tts.Communicate(
        text, VOICE,
        rate=rate_str,
        pitch=p["pitch"]
    )
    try:
        await asyncio.wait_for(
            comm.save(out_file),
            timeout=30
        )
    except asyncio.TimeoutError:
        log(
            "ERROR",
            "Edge TTS timeout 30s"
        )
        raise

# TTS — XTTS (клонирование голоса)
# ==========================================
def gen_speech_xtts(text, out_file):
    global xtts_model
    if xtts_model is None:
        log(
            "WARN",
            "XTTS not ready, "
            "falling back to Edge"
        )
        return False
    if not os.path.exists(XTTS_SAMPLE_PATH):
        log(
            "WARN",
            "sample.wav not found, "
            "falling back to Edge"
        )
        return False
    try:
        if os.path.exists(out_file):
            try:
                os.remove(out_file)
            except OSError:
                pass
        log(
            "TTS",
            f"XTTS generating: "
            f"{text[:50]}..."
        )
        clean_text = text.strip()
        if not clean_text:
            return False
        xtts_model.tts_to_file(
            text=clean_text,
            speaker_wav=XTTS_SAMPLE_PATH,
            language="ru",
            file_path=out_file
        )
        if os.path.exists(out_file):
            log(
                "TTS",
                f"XTTS done: {out_file}"
            )
            return True
        else:
            log(
                "ERROR",
                "XTTS output file "
                "not created"
            )
            return False
    except Exception as e:
        log(
            "ERROR",
            f"XTTS generation error: {e}"
        )
        return False

# УНИВЕРСАЛЬНАЯ ГЕНЕРАЦИЯ РЕЧИ
# ==========================================
def gen_speech(text, emotion, out_file):
    engine = user_settings.get(
        "tts_engine", "edge"
    )
    if engine == "xtts" and xtts_model:
        wav_file = out_file.replace(
            ".mp3", ".wav"
        )
        if not wav_file.endswith(".wav"):
            wav_file = out_file + ".wav"
        success = gen_speech_xtts(
            text, wav_file
        )
        if success:
            return wav_file
        else:
            log(
                "WARN",
                "XTTS failed, "
                "using Edge TTS"
            )
            asyncio.run(
                gen_speech_edge(
                    text, emotion, out_file
                )
            )
            return out_file
    else:
        asyncio.run(
            gen_speech_edge(
                text, emotion, out_file
            )
        )
        return out_file

def speak_chunk(audio_file):
    if stop_speaking_flag.is_set():
        return False
    if not os.path.exists(audio_file):
        log(
            "WARN",
            f"Audio file not found: "
            f"{audio_file}"
        )
        return False
    try:
        pygame.mixer.music.load(audio_file)
        pygame.mixer.music.play()
        while pygame.mixer.music.get_busy():
            if stop_speaking_flag.is_set():
                pygame.mixer.music.stop()
                pygame.mixer.music.unload()
                return False
            pygame.time.wait(50)
        pygame.mixer.music.unload()
        return True
    except Exception as e:
        log(
            "ERROR",
            f"Chunk play error: {e}"
        )
        return False

def speak(text, emotion="neutral"):
    global bot_is_speaking
    # Если играет музыка — временно тише
    music_was_playing = (
        music_state["playing"]
        and not music_state["paused"]
    )
    if music_was_playing:
        try:
            pygame.mixer.music.set_volume(
                music_state.get(
                    "volume", 0.5
                ) * 0.2
            )
        except Exception:
            pass

    with speak_lock:
        bot_is_speaking = True
        stop_speaking_flag.clear()
        try:
            ct = clean_speech(text)
            if not ct:
                bot_is_speaking = False
                return
            ct = add_speech_pauses(ct)
            chunks = (
                text_to_speech_chunks(ct)
            )
            engine = user_settings.get(
                "tts_engine", "edge"
            )
            if engine == "xtts":
                speak_xtts_mode(
                    chunks, emotion
                )
            else:
                if len(chunks) <= 1:
                    speak_single_edge(
                        ct, emotion
                    )
                else:
                    speak_streamed_edge(
                        chunks, emotion
                    )
        except Exception as e:
            log(
                "ERROR",
                f"Speech error: {e}"
            )
        finally:
            bot_is_speaking = False
            cleanup_audio_chunks()

    # Восстанавливаем громкость музыки
    if music_was_playing:
        try:
            pygame.mixer.music.set_volume(
                music_state.get(
                    "volume", 0.5
                )
            )
        except Exception:
            pass

# EDGE TTS — ОДИНОЧНЫЙ И СТРИМИНГ
# ==========================================
def speak_single_edge(text, emotion):
    try:
        if os.path.exists(AUDIO_FILE):
            os.remove(AUDIO_FILE)
    except OSError:
        pass
    try:
        asyncio.run(
            gen_speech_edge(
                text, emotion, AUDIO_FILE
            )
        )
    except Exception as e:
        log(
            "ERROR",
            f"Edge TTS gen error: {e}"
        )
        return
    speak_chunk(AUDIO_FILE)

def speak_streamed_edge(chunks, emotion):
    os.makedirs(AUDIO_DIR, exist_ok=True)
    first_file = os.path.join(
        AUDIO_DIR, "chunk_0.mp3"
    )
    _generate_edge_chunk(
        chunks[0], emotion, first_file
    )
    if (
        not os.path.exists(first_file)
        or stop_speaking_flag.is_set()
    ):
        return
    gen_thread = None
    next_file = None
    for i in range(len(chunks)):
        if stop_speaking_flag.is_set():
            break
        if i == 0:
            cur_file = first_file
        else:
            if gen_thread is not None:
                gen_thread.join(timeout=15)
            cur_file = next_file
            if (
                cur_file is None
                or not os.path.exists(
                    cur_file
                )
                or stop_speaking_flag.is_set()
            ):
                break
        if i + 1 < len(chunks):
            next_file = os.path.join(
                AUDIO_DIR,
                f"chunk_{i + 1}.mp3"
            )
            gen_thread = threading.Thread(
                target=_generate_edge_chunk,
                args=(
                    chunks[i + 1],
                    emotion,
                    next_file
                ),
                daemon=True
            )
            gen_thread.start()
        if not speak_chunk(cur_file):
            break

def _generate_edge_chunk(
    text, emotion, out_file
):
    try:
        if os.path.exists(out_file):
            os.remove(out_file)
    except OSError:
        pass
    try:
        asyncio.run(
            gen_speech_edge(
                text, emotion, out_file
            )
        )
    except Exception as e:
        log(
            "ERROR",
            f"Edge chunk gen error: {e}"
        )

# ==========================================
def speak_xtts_mode(chunks, emotion):
    os.makedirs(
        XTTS_CHUNKS_DIR, exist_ok=True
    )
    for i, chunk in enumerate(chunks):
        if stop_speaking_flag.is_set():
            break
        out_file = os.path.join(
            XTTS_CHUNKS_DIR,
            f"xtts_{i}.wav"
        )
        success = gen_speech_xtts(
            chunk, out_file
        )
        if not success:
            edge_file = os.path.join(
                XTTS_CHUNKS_DIR,
                f"edge_{i}.mp3"
            )
            try:
                asyncio.run(
                    gen_speech_edge(
                        chunk,
                        emotion,
                        edge_file
                    )
                )
                speak_chunk(edge_file)
            except Exception as e:
                log(
                    "ERROR",
                    f"Fallback edge "
                    f"chunk error: {e}"
                )
            continue
        if stop_speaking_flag.is_set():
            break
        speak_chunk(out_file)
    _cleanup_xtts_chunks()

        pass

def recognize_with_vosk(audio):
    if vosk_model is None:
        return None
    try:
        wav = audio.get_wav_data(
            convert_rate=16000,
            convert_width=2
        )
        rec = vosk.KaldiRecognizer(
            vosk_model, 16000
        )
        chunk_size = 4096
        for i in range(
            44, len(wav), chunk_size
        ):
            chunk = wav[i:i + chunk_size]
            rec.AcceptWaveform(chunk)
        result = json.loads(
            rec.FinalResult()
        )
        text = result.get(
            "text", ""
        ).strip()
        if text:
            log("STT", f"Vosk: {text}")
            return text
        return None
    except Exception as e:
        log("WARN", f"Vosk error: {e}")
        return None

def recognize_with_google(audio):
    try:
        text = recognizer.recognize_google(
            audio, language="ru-RU"
        )
        log("STT", f"Google: {text}")
        return text
    except sr.UnknownValueError:
        return None
    except sr.RequestError as e:
        log(
            "ERROR",
            f"Google STT error: {e}"
        )
        return None

def listen_once():
    with microphone as source:
        recognizer.adjust_for_ambient_noise(
            source, duration=0.5
        )
        try:
            audio = recognizer.listen(
                source,
                timeout=15,
                phrase_time_limit=30
            )
        except sr.WaitTimeoutError:
            return None
    result = recognize_with_vosk(audio)
    if result:
        return result
    return recognize_with_google(audio)

    def save_and_close(self):
        user_settings["user_name"] = (
            self.uname_var.get().strip()
            or "Dmitriy"
        )
        user_settings["user_short"] = (
            self.ushort_var.get().strip()
            or "Dima"
        )
        try:
            smin = int(self.smin_var.get())
        except ValueError:
            smin = SILENCE_MIN
        try:
            smax = int(self.smax_var.get())
        except ValueError:
            smax = SILENCE_MAX
        if smin > smax:
            smin, smax = smax, smin
        user_settings["silence_min"] = (
            max(300, smin)
        )
        user_settings["silence_max"] = max(
            user_settings["silence_min"] + 60,
            smax
        )
        user_settings[
            "speech_rate_offset"
        ] = max(
            -50,
            min(50, int(self.rate_var.get()))
        )
        user_settings["tts_engine"] = (
            self.tts_var.get()
        )
        user_settings["theme"] = (
            self.theme_var.get()
        )
        # Музыка
        new_dir = (
            self.music_dir_var.get().strip()
        )
        if new_dir:
            user_settings["music_dir"] = (
                new_dir
            )
        user_settings["music_volume"] = (
            round(self.vol_var.get(), 2)
        )
        music_state["volume"] = (
            user_settings["music_volume"]
        )
        try:
            pygame.mixer.music.set_volume(
                music_state["volume"]
            )
        except Exception:
            pass

        save_settings_file()
        log(
            "SETTINGS",
            f"Saved: {user_settings}"
        )
        if self.app_ref:
            self.app_ref.refresh_info_panel()

        msg_win = ctk.CTkToplevel(self.win)
        msg_win.title("Сохранено")
        msg_win.geometry("360x150")
        msg_win.resizable(False, False)
        msg_win.grab_set()
        ctk.CTkLabel(
            msg_win,
            text=(
                "Настройки сохранены!\n"
                "Тема и движок TTS\n"
                "применятся при перезапуске."
            ),
            font=ctk.CTkFont(size=13)
        ).pack(pady=(20, 10))
        ctk.CTkButton(
            msg_win,
            text="OK",
            width=80,
            command=msg_win.destroy
        ).pack()
        self.win.destroy()

class App:
    def __init__(self):
        self.root = ctk.CTk()
        self.root.title(
            f"{COMPANION_NAME} v{APP_VERSION}"
        )
        theme_key = user_settings.get(
            "theme", "dark_minimal"
        )
        self.th = THEMES.get(
            theme_key,
            THEMES["dark_minimal"]
        )
        sw = self.root.winfo_screenwidth()
        sh = self.root.winfo_screenheight()
        ww, wh = 1100, 700
        xx = (sw - ww) // 2
        yy = (sh - wh) // 2
        self.root.geometry(
            f"{ww}x{wh}+{xx}+{yy}"
        )
        self.root.resizable(True, True)
        self.root.protocol(
            "WM_DELETE_WINDOW",
            self.on_close
        )
        self.running = True
        self.listening = False
        self.auto = False
        self._is_thinking = False
        self.night_mode = True
        self.avatars = {}
        self.avimg = None
        self.avatar_label = None
        self.session_msgs = 0
        self.last_msg_date = None
        self.mic_available = True
        self.info_labels = []
        self.build_ui()

    def on_close(self):
        self.running = False
        self.auto = False
        stop_speaking_flag.set()
        music_stop()
        t0 = time.time()
        while (
            bot_is_speaking
            and time.time() - t0 < 3
        ):
            time.sleep(0.1)
        try:
            try:
                memory_db.summarize_session(
                    history_dicts
                )
            except Exception:
                pass
            save_memory()
            save_facts()
            save_events()
            save_mood(current_mood)
            save_session()
            token_tracker.save()
            closeness.save()
            save_settings_file()
            cleanup_audio_chunks()
            log(
                "INFO",
                "Application closed"
            )
            pygame.mixer.quit()
        except Exception:
            pass
        self.root.destroy()

    def load_avatars(self):
        ad = os.path.join(
            PROJECT_DIR, "avatars"
        )
        if os.path.exists(ad):
            for emo in EMOTIONS_VOICE.keys():
                for ext in [
                    ".png", ".jpg", ".jpeg"
                ]:
                    p = os.path.join(
                        ad, emo + ext
                    )
                    if os.path.exists(p):
                        im = Image.open(p)
                        im = im.resize(
                            (260, 260),
                            Image.LANCZOS
                        )
                        self.avatars[emo] = (
                            ImageTk.PhotoImage(
                                im
                            )
                        )
                        break
        if not self.avatars:
            for ext in [
                ".png", ".jpg",
                ".jpeg", ".webp"
            ]:
                ap = os.path.join(
                    PROJECT_DIR,
                    "avatar" + ext
                )
                if os.path.exists(ap):
                    img = Image.open(ap)
                    img = img.resize(
                        (260, 260),
                        Image.LANCZOS
                    )
                    default = (
                        ImageTk.PhotoImage(
                            img
                        )
                    )
                    for emo in (
                        EMOTIONS_VOICE.keys()
                    ):
                        self.avatars[emo] = (
                            default
                        )
                    break
        if self.avatars:
            self.avimg = self.avatars.get(
                "neutral",
                list(
                    self.avatars.values()
                )[0]
            )

    def _update_music_label(self):

    def msg(
        self, who, text, tag,
        show_time=True
    ):
        self.chat.configure(state="normal")
        th = self.th
        if show_time:
            today = (
                datetime.now().strftime(
                    "%d.%m.%Y"
                )
            )
            if self.last_msg_date != today:
                if (
                    self.last_msg_date
                    is not None
                ):
                    days_ru = [
                        "Понедельник",
                        "Вторник",
                        "Среда",
                        "Четверг",
                        "Пятница",
                        "Суббота",
                        "Воскресенье"
                    ]
                    wd = days_ru[
                        datetime.now()
                        .weekday()
                    ]
                    self.chat.insert(
                        "end",
                        f"\n──── {today} "
                        f"({wd}) ────\n\n"
                    )
                self.last_msg_date = today
            timestamp = (
                datetime.now().strftime(
                    "[%H:%M]"
                )
            )
            prefix = f"{timestamp} "
        else:
            prefix = ""
        color_map = {
            "u": th["user_color"],
            "b": th["bot_color"],
            "s": th["sys_color"],
        }
        color = color_map.get(
            tag, th["text"]
        )
        self.chat.insert(
            "end",
            f"{prefix}{who}: {text}\n\n"
        )
        try:
            line_start = (
                self.chat.index(
                    "end-3l linestart"
                )
            )
            line_end = (
                self.chat.index(
                    "end-2l lineend"
                )
            )
            tag_name = f"color_{tag}"
            self.chat.tag_config(
                tag_name,
                foreground=color
            )
            self.chat.tag_add(
                tag_name,
                line_start,
                line_end
            )
        except Exception:
            pass
        self.chat.see("end")
        self.chat.configure(state="disabled")

    def is_night_now(self):
        if not self.night_mode:
            return False
        h = datetime.now().hour
        if NIGHT_START > NIGHT_END:
            return (
                h >= NIGHT_START
                or h < NIGHT_END
            )
        else:
            return (
                NIGHT_START <= h < NIGHT_END
            )

                    .strftime("%Y%m%d_%H%M")
                    + ".txt"
                )
                content = self.chat.get(
                    "1.0", "end"
                )
                with open(
                    export_path, "w",
                    encoding="utf-8"
                ) as f:
                    f.write(content)
                self.msg(
                    "System",
                    f"Чат сохранён:\n"
                    f"{export_path}",
                    "s"
                )
            except Exception as e:
                self.msg(
                    "System",
                    f"Ошибка: {e}",
                    "s"
                )
            return True
        return False

    def on_send(self, event=None):
        text = self.inp.get().strip()
        if not text:
            return
        self.inp.delete(0, "end")
        self._update_char_count()
        if text.startswith("/"):
            if self.handle_command(text):
                return
        if len(text) > MAX_USER_MESSAGE:
            text = text[:MAX_USER_MESSAGE]
            self.msg(
                "System",
                f"(обрезано до "
                f"{MAX_USER_MESSAGE} симв.)",
                "s"
            )
        self.msg("Вы", text, "u")
        threading.Thread(
            target=self.process,
            args=(text,),
            daemon=True
        ).start()

    def on_voice(self):
        if (
            self.listening
            or bot_is_speaking
            or not self.mic_available
        ):
            return
        threading.Thread(
            target=self.vlisten,
            daemon=True
        ).start()

    def vlisten(self):
        self.listening = True
        self.root.after(

        )
        self.root.after(
            0,
            lambda: self.vbtn.configure(

        )

        self.listening = False

                "#ff4444"
            )
            return
        self.root.after(
            0, self.msg,
            "Вы", heard, "u"
        )
        self.process(heard)

    def toggle_auto(self):
        if not self.mic_available:
            self.root.after(
                0, self.status,
                "Микрофон недоступен",
                "#ff4444"
            )
            return
        self.auto = not self.auto
        th = self.th
        if self.auto:
            self.abtn.configure(
                text="Авто: ВКЛ",
                fg_color=th["btn_auto_on"]
            )
            threading.Thread(
                target=self.auto_loop,
                daemon=True
            ).start()
        else:
            self.abtn.configure(
                text="Авто: ВЫКЛ",
                fg_color=th["btn_auto_off"]
            )
            self.root.after(
                0, self.status,
                "Готово", "#00ff88"
            )

    def auto_loop(self):
        while self.auto and self.running:
            if bot_is_speaking:
                time.sleep(0.5)
                continue
            self.listening = True
            self.root.after(
                0, self.status,
                "🎤 Слушаю...", "#ffaa00"
            )
            heard = listen_once()
            self.listening = False
            if (
                not self.auto
                or not self.running
            ):
                break
            if heard is None:
                self.root.after(
                    0, self.status,
                    "🎤 Слушаю...", "#00ff88"
                )
                continue
            self.root.after(
                0, self.msg,
                "Вы", heard, "u"
            )
            self.process(heard)

    def _typing_animation(self):
        dots = ["", ".", "..", "..."]
        i = 0
        while self._is_thinking:
            d = dots[i % 4]
            self.root.after(
                0, self.status,
                f"Юля печатает{d}",
                "#ff69b4"
            )
            time.sleep(0.4)
            i += 1

    def process(self, text):
        global last_interaction
        last_interaction = time.time()
        self._is_thinking = True
        threading.Thread(
            target=self._typing_animation,
            daemon=True
        ).start()
        try:
            raw = get_response(
                text, app_ref=self
            )
            self._is_thinking = False
            em, clean = parse_emotion(raw)
            update_mood(em)
            self.session_msgs += 1
            user_emo, _ = (
                analyze_user_emotion(text)
            )
            level_up = closeness.on_message(
                user_emo
            )
            self.root.after(
                0, self.set_emotion, em
            )
            self.root.after(
                0,
                self.update_mood_display
            )
            self.root.after(
                0,
                self.update_token_display
            )
            self.root.after(
                0,
                self.update_session_display
            )
            self.root.after(
                0,
                self.update_closeness_display
            )
            self.root.after(
                0, self.msg,
                COMPANION_NAME, clean, "b"
            )
            self.root.after(
                0, self.status,
                "Говорит...", "#ff69b4"
            )
            speak(clean, em)
            if level_up is not None:
                self._handle_level_up(
                    level_up
                )
            if self.auto:
                self.root.after(
                    0, self.status,
                    "🎤 Слушаю...", "#00ff88"
                )
            else:
                self.root.after(
                    0, self.status,
                    "Готово", "#00ff88"
                )
            last_interaction = time.time()
        except Exception as e:
            self._is_thinking = False
            err_str = str(e).lower()
            if (
                "connect" in err_str
                or "network" in err_str
            ):
                user_msg = (
                    "Нет подключения "
                    "к интернету"
                )
            elif "timeout" in err_str:
                user_msg = (
                    "Сервер не отвечает"
                )
            elif (
                "auth" in err_str
                or "401" in err_str
            ):
                user_msg = (
                    "Ошибка авторизации"
                )
            elif (
                "rate" in err_str
                or "429" in err_str
            ):
                user_msg = (
                    "Слишком много запросов"
                )
            else:
                user_msg = (
                    f"Ошибка: "
                    f"{str(e)[:80]}"
                )
            log(
                "ERROR",
                f"Process error: {e}"
            )
            self.root.after(
                0, self.status,
                user_msg, "#ff4444"
            )
            self.root.after(
                0, self.msg,
                "System", user_msg, "s"
            )

    def _handle_level_up(self, new_level):
        try:
            prompt_text = (
                LEVEL_UP_PROMPTS.get(
                    new_level
                )
            )
            if not prompt_text:
                return
            info = CLOSENESS_LEVELS.get(
                new_level,
                CLOSENESS_LEVELS[1]
            )
            self.root.after(
                0, self.msg,
                "System",
                f"💛 Уровень близости: "
                f"{info['name_display']}!",
                "s"
            )
            time.sleep(1.5)
            with history_lock:
                temp = history_dicts.copy()
                temp[0] = {
                    "role": "system",
                    "content": make_prompt()
                }
                temp.append({
                    "role": "user",
                    "content": prompt_text
                })
            raw = call_gigachat(
                temp,
                temp=0.85,
                max_tok=200
            )
            em, clean = parse_emotion(raw)
            update_mood(em)
            with history_lock:
                history_dicts.append({
                    "role": "assistant",
                    "content": raw
                })
            self.root.after(
                0, self.set_emotion, em
            )
            self.root.after(
                0,
                self.update_closeness_display
            )
            self.root.after(
                0, self.msg,
                COMPANION_NAME, clean, "b"
            )
            play_notification()
            speak(clean, em)
        except Exception as e:
            log(
                "ERROR",
                f"Level-up reaction: {e}"
            )

    def initiative_loop(self):
        global last_interaction
        wait = 300
        while wait > 0 and self.running:
            time.sleep(1)
            wait -= 1
        while self.running:
            time.sleep(10)
            if not self.running:
                break
            smin = user_settings.get(
                "silence_min", SILENCE_MIN
            )
            smax = user_settings.get(
                "silence_max", SILENCE_MAX
            )
            nxt = random.randint(smin, smax)
            elapsed = (
                time.time() - last_interaction
            )
            busy = (
                bot_is_speaking
                or self.listening
            )
            night = self.is_night_now()
            info = closeness.get_level_info()
            mult = info.get(
                "initiative_mult", 1.0
            )
            adjusted_nxt = int(
                nxt / max(0.1, mult)
            )
            if (
                elapsed > adjusted_nxt
                and not busy
                and not night
            ):
                try:
                    raw = get_initiative()
                    em, cl = parse_emotion(
                        raw
                    )
                    update_mood(em)
                    self.root.after(
                        0,
                        self.set_emotion, em
                    )
                    self.root.after(
                        0,
                        self.update_mood_display
                    )
                    self.root.after(
                        0,
                        self.update_token_display
                    )
                    self.root.after(
                        0,
                        self.update_closeness_display
                    )
                    self.root.after(
                        0, self.msg,
                        COMPANION_NAME,
                        cl, "b"
                    )
                    play_notification()
                    speak(cl, em)
                    last_interaction = (
                        time.time()
                    )
                except Exception as e:
                    log(
                        "ERROR",
                        f"Initiative "
                        f"error: {e}"
                    )

    def run(self):
        mic_ok, mic_err = (
            check_microphone()
        )
        if not mic_ok:
            self.mic_available = False
            self.msg(
                "System",
                f"Микрофон недоступен: "
                f"{mic_err}\n"
                f"Голосовой ввод отключён.",
                "s"
            )
            self.vbtn.configure(
                state="disabled",
                fg_color="gray30"
            )
        else:
            self.msg(
                "System",
                f"Микрофон: {mic_err}",
                "s"
            )

        mem_stats = memory_db.get_stats()
        if mem_stats['ready']:
            self.msg(
                "System",
                f"🧠 Память: "
                f"{mem_stats['total_memories']}"
                f" воспоминаний, "
                f"{mem_stats['summaries']}"
                f" суммаризаций",
                "s"
            )
        else:
            self.msg(
                "System",
                "🧠 Память: "
                "ChromaDB недоступен",
                "s"
            )

        tts_engine = user_settings.get(
            "tts_engine", "edge"
        )
        if tts_engine == "xtts":
            if not XTTS_AVAILABLE:
                self.msg(
                    "System",
                    "TTS: XTTS не установлен.\n"
                    "pip install TTS\n"
                    "Используется Edge TTS.",
                    "s"
                )
            elif not os.path.exists(
                XTTS_SAMPLE_PATH
            ):
                self.msg(
                    "System",
                    f"TTS: sample.wav "
                    f"не найден!\n"
                    f"Путь: {XTTS_SAMPLE_PATH}",
                    "s"
                )
            else:
                self.msg(
                    "System",
                    "TTS: XTTS загружается...",
                    "s"
                )
        else:
            self.msg(
                "System",
                "🔊 TTS: Edge TTS готов",
                "s"
            )

        # Статус музыки
        library = scan_music_library()
        total_tracks = sum(
            len(v) for v in library.values()
        )
        mdir = user_settings.get(
            "music_dir", MUSIC_DIR
        )
        if total_tracks > 0:
            self.msg(
                "System",
                f"🎵 Музыка: {total_tracks} "
                f"треков в {len(library)} "
                f"жанрах\n"
                f"Папка: {mdir}",
                "s"
            )
        else:
            self.msg(
                "System",
                f"🎵 Музыка: папка пуста\n"
                f"{mdir}",
                "s"
            )

        theme_name = THEMES.get(
            user_settings.get(
                "theme", "dark_minimal"
            ),
            THEMES["dark_minimal"]
        )["name"]
        self.msg(
            "System",
            f"🎨 Тема: {theme_name}",
            "s"
        )

        self.show_history()
        threading.Thread(
            target=self.greet,
            daemon=True
        ).start()
        threading.Thread(
            target=self.initiative_loop,
            daemon=True
        ).start()
        self.root.mainloop()
